#!groovy
@Library('hydro_groovy') _
def deploy = new devops.ci()
pipeline {
agent {
label 'make'
}
environment {
repository = "https://xxxx/aa.git"
client_common_repository = "https://xxxx/bbb.git"
GIT_COMMIT = "UNKNOW"
Previous_version = "UNKNOW"
port = 6023
}
stages {
stage("deleteDir"){
when {
environment name: 'deleteDir', value: 'yes'
}
steps{
deleteDir()
}
}
stage("buildInfo"){
steps{
script{
deploy.buildInfo(String[] args)
}
}
}
stage("checkout"){
when {
environment name: 'submodule', value: 'no'
}
steps{
script{
deploy.checkout_no_submodule(String[] args)
}
}
}
stage("checkout&submodule"){
when {
environment name: 'submodule', value: 'yes'
}
steps{
script{
deploy.checkout(String[] args)
}
}
}
stage("checkout_client_common"){
steps{
script{
deploy.checkout_client_common(String[] args)
}
}
}
stage("make"){
steps{
script{
deploy.make_tgz(String[] args)
}
}
}
stage("push tgz"){
when {
environment name: 'only_make', value: 'no'
}
steps{
script{
deploy.push_tgz(String[] args)
}
}
}
stage("check_version"){
when {
environment name: 'only_make', value: 'no'
}
steps{
script{
deploy.check_version(String[] args)
}
}
}
stage("upgrade"){
when {
environment name: 'only_make', value: 'no'
}
steps{
script{
deploy.upgrade_stor(String[] args)
}
}
}
}
post {
success {
script {
deploy.success(String[] args)
}
}
failure {
script {
deploy.failure(String[] args)
}
}
aborted {
script {
deploy.aborted(String[] args)
}
}
}
}
